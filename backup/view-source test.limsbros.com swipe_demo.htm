#first {
width:100%;
height:100%;
}
#second {
height:100%;
position:absolute;
background-color:yellowgreen;
}
#third {
height:100%;
position:absolute;
background-color:green;
}
</style>
<script type='text/javascript' src='http://ajax.googleapis.com/ajax/libs/jquery/1.4/jquery.min.js'></script>
<script type="text/javascript">
var down_x = null;
var up_x = null;
var mouse_down = false;
var speed = 300;
$().ready(function(){
var first = $("#first");
var pos = first.position();
var winW = $(window).width();
$("#second").css('top', pos.top).css('left', winW-200).css('width', first.width()-200);
$("#third").css('top', pos.top).css('left', winW-100).css('width', first.width()-300);
//register swipe
register_swipe($("#second"), second_swipe);
register_swipe($("#third"), third_swipe);
});
function second_swipe()
{
if ((down_x - up_x) > 50)
{
second_swipe_left();
}
else if ((up_x - down_x) > 50)
{
second_swipe_right();
third_swipe_right();
}
}
function second_swipe_left()
{
$("#second").animate({left:200},speed);
}
function second_swipe_right()
{
var winW = $(window).width();
$("#second").animate({left:winW-200},speed);
}
function third_swipe()
{
if ((down_x - up_x) > 50)
{
second_swipe_left();
third_swipe_left();
}
else if ((up_x - down_x) > 50)
{
third_swipe_right();
}
}
function third_swipe_left()
{
$("#third").animate({left:300},speed);
}
function third_swipe_right()
{
var winW = $(window).width();
$("#third").animate({left:winW-100},speed);
}
function register_swipe(div, callback)
{
div.mousedown(function(e){
e.preventDefault();
down_x = e.pageX;
up_x = e.pageX;
mouse_down = true;
$("body").unbind();
$("body").mousemove(function(e){
if (mouse_down)
{
var diff = e.pageX - up_x;
var left = parseInt(div.css('left').replace('px',''));
div.css('left',left+diff);
up_x = e.pageX;
}
});
$("body").mouseup(function(e){
up_x = e.pageX;
callback();
$(this).unbind();
mouse_down = false;
});
});
div.bind('touchstart', function(e){
down_x = e.originalEvent.touches[0].pageX;
up_x = down_x;
$("body").unbind();
$("body").bind('touchmove', function(e){
e.preventDefault();
var diff = e.originalEvent.touches[0].pageX - up_x;
var left = parseInt(div.css('left').replace('px',''));
div.css('left',left+diff);
up_x = e.originalEvent.touches[0].pageX;
});
$("body").bind('touchend', function(e){
callback();
$(this).unbind();
});
});
}
</script>
